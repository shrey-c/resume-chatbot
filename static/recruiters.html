Layer 3: Prompt Injection Prevention

BLOCKED_PATTERNS = [
    r"ignore (previous|all) (instructions|prompts)",
    r"system:",
    r"assistant:",
    r"<\|im_start\|>",
    # ... more patterns
]

def detect_prompt_injection(user_input: str) -> bool:
    """Detect potential prompt injection attacks."""
    for pattern in BLOCKED_PATTERNS:
        if re.search(pattern, user_input, re.IGNORECASE):
            return True
    return False

Layer 4: JWT Authentication (Admin Portal)

# Password hashing with bcrypt
def generate_password_hash(password: str) -> str:
    return bcrypt.hashpw(
        password.encode('utf-8'), 
        bcrypt.gensalt()
    ).decode('utf-8')

# JWT token generation
def create_access_token(data: dict):
    to_encode = data.copy()
    expire = datetime.utcnow() + timedelta(hours=24)
    to_encode.update({"exp": expire})
    return jwt.encode(to_encode, SECRET_KEY, algorithm="HS256")

Agent Implementation Details

class ResearchAgent:
    """Analyzes queries and extracts relevant resume context."""
    
    def __init__(self):
        self.llm = ChatOllama(
            model="llama3.2:1b",
            temperature=0.3  # Lower temp = more focused
        )
    
    async def analyze(self, state: AgentState) -> AgentState:
        # Prioritize HTML context
        context = state.get("html_context") or state.get("resume_context")
        
        prompt = f"""Analyze: {state['user_query']}
        Context: {context}
        Extract relevant information."""
        
        result = await self.llm.ainvoke(prompt)
        state["research_results"] = result.content
        return state

HTML-First Context Loading

def get_html_context() -> str:
    """
    PRIMARY source for chatbot responses.
    Extracts resume data from HTML using BeautifulSoup4.
    """
    parser = HTMLResumeParser("static/index.html")
    
    # Prioritize static data section (pre-rendered)
    static_data = parser.soup.find('div', id='resume-static-data')
    if static_data:
        return parser.extract_from_static(static_data)
    
    # Fallback: extract from dynamic sections
    return parser.extract_from_dynamic_sections()

async def chat(self, message: str) -> str:
    state = {
        "user_query": message,
        "html_context": get_html_context(),      # PRIMARY
        "resume_context": get_resume_context(),  # FALLBACK
    }
    result = await self.workflow.ainvoke(state)
    return result["final_response"]
